import React, { useState, useEffect, useMemo } from 'react';
import { 
  Copy, 
  ExternalLink, 
  ChevronDown, 
  ChevronUp, 
  Zap, 
  Layout, 
  Type, 
  Image as ImageIcon,
  CheckCircle2,
  Info,
  Layers,
  Sparkles,
  MousePointer2,
  ArrowRight
} from 'lucide-react';

const App = () => {
  const [formData, setFormData] = useState({
    goal: '',
    audience: '',
    platform: 'Instagram',
    tone: 'Premium (Prabangus)',
    object: '',
    environment: '',
    camera: 'Close-up (Stambus planas)',
    lighting: 'Cinematic (Kino apšvietimas)',
    color: '',
    style: 'Tikroviška nuotrauka',
    headline: '',
    subline: '',
    cta: '',
    textPosition: 'Centras',
    typography: 'Modernus Sans-serif',
    contrast: 'Aukštas kontrastas'
  });

  const [copied, setCopied] = useState(false);
  const [activeStep, setActiveStep] = useState(1);
  const [showTips, setShowTips] = useState(false);
  const [showExamples, setShowExamples] = useState(false);

  const generatedPrompt = useMemo(() => {
    const { 
      goal, audience, platform, tone, 
      object, environment, camera, lighting, color, style,
      headline, subline, cta, textPosition, typography, contrast 
    } = formData;

    let promptParts = [];
    const stylePrefix = style ? `${style}, ` : 'Profesionalus vaizdas: ';
    const coreVisual = `${object || '[Objektas]'} ${environment ? `${environment} aplinkoje` : ''}`;
    promptParts.push(`${stylePrefix}${coreVisual}.`);
    
    const technical = `${camera}, ${lighting}, ${color ? `${color} spalvų gama, ` : ''}itin detalu, aukščiausia kokybė, profesionali kompozicija.`;
    promptParts.push(technical);

    if (goal || audience || platform) {
      promptParts.push(`Sukurta rinkodaros tikslui: ${goal || 'reklama'}, skirta ${audience || 'tikslinė auditorija'} platformai ${platform}. Nuotaika: ${tone}.`);
    }

    if (headline || subline || cta) {
      const textSection = `Reklaminis maketas su integruotu tekstu: "${headline}" ${subline ? `ir paantrašte "${subline}"` : ''} ${cta ? `su kvietimu veikti "${cta}"` : ''}. Šrifto stilius: ${typography}. Teksto pozicija: ${textPosition}. Reikalavimas: ${contrast}, švari erdvė aplink tekstą skaitomumui užtikrinti.`;
      promptParts.push(textSection);
    }

    return promptParts.join(' ');
  }, [formData]);

  const handleInputChange = (e) => {
    const { name, value } = e.target;
    setFormData(prev => ({ ...prev, [name]: value }));
  };

  const copyToClipboard = () => {
    const el = document.createElement('textarea');
    el.value = generatedPrompt;
    document.body.appendChild(el);
    el.select();
    document.execCommand('copy');
    document.body.removeChild(el);
    setCopied(true);
    setTimeout(() => setCopied(false), 2000);
  };

  const openTool = (url) => {
    copyToClipboard();
    window.open(url, '_blank');
  };

  const tools = [
    { name: 'Ideogram', url: 'https://ideogram.ai/', desc: 'Geriausias tekstui', icon: '🎨' },
    { name: 'Leonardo.ai', url: 'https://leonardo.ai/', desc: 'Maksimali kontrolė', icon: '⚡' },
    { name: 'Midjourney', url: 'https://www.midjourney.com/', desc: 'Aukščiausia kokybė', icon: '⛵' },
    { name: 'ChatGPT / DALL-E', url: 'https://chat.openai.com/', desc: 'Greitas ir paprastas', icon: '🤖' },
    { name: 'Adobe Firefly', url: 'https://firefly.adobe.com/', desc: 'Komerciniam naudojimui', icon: '🖌️' },
    { name: 'Canva Magic', url: 'https://www.canva.com/', desc: 'Dizaino integracija', icon: '✨' }
  ];

  return (
    <div className="min-h-screen bg-[#F8FAFC] text-slate-900 font-sans selection:bg-indigo-100">
      {/* Premium Header */}
      <header className="bg-white/80 backdrop-blur-md border-b border-slate-200 sticky top-0 z-50">
        <div className="max-w-7xl mx-auto px-6 h-20 flex items-center justify-between">
          <div className="flex items-center gap-3">
            <div className="bg-gradient-to-br from-indigo-600 to-violet-700 p-2 rounded-xl shadow-lg shadow-indigo-200">
              <Sparkles className="w-6 h-6 text-white" />
            </div>
            <div>
              <h1 className="font-extrabold text-xl tracking-tight">DI Generatorius</h1>
              <p className="text-[10px] text-indigo-600 font-bold uppercase tracking-widest leading-none">Rinkodaros Specialistams</p>
            </div>
          </div>
          
          <nav className="hidden lg:flex items-center gap-1 bg-slate-100 p-1 rounded-full border border-slate-200">
            {['Kontekstas', 'Vizualas', 'Tekstas', 'Eksportas'].map((step, idx) => (
              <div 
                key={step}
                className={`px-4 py-1.5 rounded-full text-xs font-bold transition-all ${activeStep === idx + 1 ? 'bg-white text-indigo-600 shadow-sm' : 'text-slate-500'}`}
              >
                {idx + 1}. {step}
              </div>
            ))}
          </nav>

          <button 
            onClick={copyToClipboard}
            className="hidden sm:flex items-center gap-2 bg-slate-900 text-white px-5 py-2.5 rounded-xl text-sm font-bold hover:bg-slate-800 transition-all active:scale-95"
          >
            {copied ? <CheckCircle2 className="w-4 h-4 text-green-400" /> : <Copy className="w-4 h-4" />}
            {copied ? 'Nukopijuota' : 'Kopijuoti Promptą'}
          </button>
        </div>
      </header>

      <main className="max-w-7xl mx-auto px-6 py-10">
        <div className="grid grid-cols-1 lg:grid-cols-12 gap-10">
          
          {/* Form Side */}
          <div className="lg:col-span-7 space-y-10">
            
            {/* Step 1: Campaign Context */}
            <section className="bg-white p-8 rounded-3xl shadow-sm border border-slate-200 hover:border-indigo-200 transition-colors" onFocus={() => setActiveStep(1)}>
              <div className="flex items-center gap-4 mb-8">
                <div className="w-10 h-10 bg-indigo-50 rounded-xl flex items-center justify-center">
                  <Layout className="w-5 h-5 text-indigo-600" />
                </div>
                <h2 className="text-xl font-extrabold">1. Kampanijos kontekstas</h2>
              </div>
              <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
                <div className="space-y-2">
                  <label className="text-[11px] font-black text-slate-400 uppercase tracking-wider">Kampanijos tikslas</label>
                  <input 
                    type="text" 
                    name="goal"
                    placeholder="Pvz.: Naujo produkto pristatymas"
                    className="w-full p-4 bg-slate-50 border border-slate-100 rounded-2xl focus:bg-white focus:ring-4 focus:ring-indigo-500/10 focus:border-indigo-500 outline-none transition-all font-medium"
                    onChange={handleInputChange}
                  />
                </div>
                <div className="space-y-2">
                  <label className="text-[11px] font-black text-slate-400 uppercase tracking-wider">Tikslinė platforma</label>
                  <select 
                    name="platform"
                    className="w-full p-4 bg-slate-50 border border-slate-100 rounded-2xl outline-none appearance-none cursor-pointer font-medium"
                    onChange={handleInputChange}
                  >
                    <option>Instagram</option>
                    <option>LinkedIn Postas</option>
                    <option>Facebook Reklama</option>
                    <option>Web Hero Baneris</option>
                    <option>Lauko reklama (Print)</option>
                  </select>
                </div>
                <div className="space-y-2">
                  <label className="text-[11px] font-black text-slate-400 uppercase tracking-wider">Auditorija</label>
                  <input 
                    type="text" 
                    name="audience"
                    placeholder="Pvz.: Jaunos šeimos, Gen Z"
                    className="w-full p-4 bg-slate-50 border border-slate-100 rounded-2xl outline-none font-medium"
                    onChange={handleInputChange}
                  />
                </div>
                <div className="space-y-2">
                  <label className="text-[11px] font-black text-slate-400 uppercase tracking-wider">Prekės ženklo tonas</label>
                  <select 
                    name="tone"
                    className="w-full p-4 bg-slate-50 border border-slate-100 rounded-2xl outline-none font-medium"
                    onChange={handleInputChange}
                  >
                    <option>Premium (Prabangus)</option>
                    <option>Bold (Drąsus)</option>
                    <option>Minimalistinis</option>
                    <option>Žaismingas</option>
                    <option>Ekspertiškas</option>
                  </select>
                </div>
              </div>
            </section>

            {/* Step 2: Visual Core */}
            <section className="bg-white p-8 rounded-3xl shadow-sm border border-slate-200" onFocus={() => setActiveStep(2)}>
              <div className="flex items-center gap-4 mb-8">
                <div className="w-10 h-10 bg-indigo-50 rounded-xl flex items-center justify-center">
                  <ImageIcon className="w-5 h-5 text-indigo-600" />
                </div>
                <h2 className="text-xl font-extrabold">2. Vizualo esmė</h2>
              </div>
              <div className="space-y-6">
                <div className="space-y-2">
                  <label className="text-[11px] font-black text-slate-400 uppercase tracking-wider">Kas vaizduojama (Pagrindinis objektas)</label>
                  <textarea 
                    name="object"
                    rows="2"
                    placeholder="Pvz.: Prabangus rankinis laikrodis ant marmurinio stalo"
                    className="w-full p-4 bg-slate-50 border border-slate-100 rounded-2xl outline-none resize-none font-medium"
                    onChange={handleInputChange}
                  />
                </div>
                <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
                  <div className="space-y-2">
                    <label className="text-[11px] font-black text-slate-400 uppercase tracking-wider">Stilius</label>
                    <select name="style" className="w-full p-4 bg-slate-50 border border-slate-100 rounded-2xl outline-none font-medium" onChange={handleInputChange}>
                      <option>Tikroviška nuotrauka</option>
                      <option>3D Renderis (Studio)</option>
                      <option>Kinematografinis stilius</option>
                      <option>Mados žurnalo stilius</option>
                      <option>Minimalistinė iliustracija</option>
                    </select>
                  </div>
                  <div className="space-y-2">
                    <label className="text-[11px] font-black text-slate-400 uppercase tracking-wider">Apšvietimas</label>
                    <select name="lighting" className="w-full p-4 bg-slate-50 border border-slate-100 rounded-2xl outline-none font-medium" onChange={handleInputChange}>
                      <option>Cinematic (Kino apšvietimas)</option>
                      <option>Minkšta dienos šviesa</option>
                      <option>Auksinė valanda (Golden Hour)</option>
                      <option>Studijinis apšvietimas</option>
                      <option>Neoninis apšvietimas</option>
                    </select>
                  </div>
                  <div className="space-y-2">
                    <label className="text-[11px] font-black text-slate-400 uppercase tracking-wider">Kameros kampas</label>
                    <select name="camera" className="w-full p-4 bg-slate-50 border border-slate-100 rounded-2xl outline-none font-medium" onChange={handleInputChange}>
                      <option>Close-up (Stambus planas)</option>
                      <option>Akių lygis (Eye level)</option>
                      <option>Iš viršaus (Flatlay)</option>
                      <option>Platus kampas (Wide angle)</option>
                      <option>Iš apačios (Hero shot)</option>
                    </select>
                  </div>
                  <div className="space-y-2">
                    <label className="text-[11px] font-black text-slate-400 uppercase tracking-wider">Dominuojanti spalva</label>
                    <input 
                      type="text" 
                      name="color"
                      placeholder="Pvz.: Švelni pastelinė, Gili mėlyna"
                      className="w-full p-4 bg-slate-50 border border-slate-100 rounded-2xl outline-none font-medium"
                      onChange={handleInputChange}
                    />
                  </div>
                </div>
              </div>
            </section>

            {/* Step 3: Marketing Text Integration */}
            <section className="bg-white p-8 rounded-3xl shadow-sm border border-slate-200" onFocus={() => setActiveStep(3)}>
              <div className="flex items-center gap-4 mb-8">
                <div className="w-10 h-10 bg-indigo-50 rounded-xl flex items-center justify-center">
                  <Type className="w-5 h-5 text-indigo-600" />
                </div>
                <h2 className="text-xl font-extrabold">3. Tekstų integracija</h2>
              </div>
              <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
                <div className="space-y-2">
                  <label className="text-[11px] font-black text-slate-400 uppercase tracking-wider">Antraštė (Headline)</label>
                  <input 
                    type="text" 
                    name="headline"
                    placeholder="Pvz.: Pajuskite prabangą"
                    className="w-full p-4 bg-slate-50 border border-slate-100 rounded-2xl outline-none font-medium"
                    onChange={handleInputChange}
                  />
                </div>
                <div className="space-y-2">
                  <label className="text-[11px] font-black text-slate-400 uppercase tracking-wider">Kvietimas veikti (CTA)</label>
                  <input 
                    type="text" 
                    name="cta"
                    placeholder="Pvz.: Pirkti dabar"
                    className="w-full p-4 bg-slate-50 border border-slate-100 rounded-2xl outline-none font-medium"
                    onChange={handleInputChange}
                  />
                </div>
                <div className="space-y-2">
                  <label className="text-[11px] font-black text-slate-400 uppercase tracking-wider">Teksto pozicija</label>
                  <select name="textPosition" className="w-full p-4 bg-slate-50 border border-slate-100 rounded-2xl outline-none font-medium" onChange={handleInputChange}>
                    <option>Centras</option>
                    <option>Viršutinė dalis</option>
                    <option>Apatinis trečdalis</option>
                    <option>Dinaminis išdėstymas</option>
                  </select>
                </div>
                <div className="space-y-2">
                  <label className="text-[11px] font-black text-slate-400 uppercase tracking-wider">Šrifto stilius</label>
                  <select name="typography" className="w-full p-4 bg-slate-50 border border-slate-100 rounded-2xl outline-none font-medium" onChange={handleInputChange}>
                    <option>Modernus Sans-serif</option>
                    <option>Prabangus Serif</option>
                    <option>Minimalistinis</option>
                    <option>Rankraštinis</option>
                  </select>
                </div>
              </div>
            </section>

          </div>

          {/* Sticky Output Side */}
          <div className="lg:col-span-5">
            <div className="sticky top-28 space-y-8">
              
              {/* Output Card */}
              <div className="bg-slate-900 rounded-[2.5rem] p-8 shadow-2xl shadow-indigo-200/50 text-white relative overflow-hidden group">
                <div className="absolute top-0 right-0 p-8">
                  <Zap className="w-12 h-12 text-white/5" />
                </div>
                
                <div className="flex items-center justify-between mb-8">
                  <div className="flex items-center gap-2">
                    <span className="w-2 h-2 bg-indigo-500 rounded-full animate-pulse"></span>
                    <span className="text-[11px] font-black uppercase tracking-[0.2em] text-indigo-400">Sugeneruotas Promptas</span>
                  </div>
                  <button 
                    onClick={copyToClipboard}
                    className="p-3 bg-white/10 hover:bg-white/20 rounded-2xl transition-all"
                    title="Kopijuoti"
                  >
                    {copied ? <CheckCircle2 className="w-5 h-5 text-green-400" /> : <Copy className="w-5 h-5" />}
                  </button>
                </div>

                <div className="min-h-[220px] text-lg leading-relaxed font-medium text-slate-100 italic selection:bg-indigo-500">
                  {generatedPrompt}
                </div>

                <div className="mt-10 pt-8 border-t border-white/10">
                  <p className="text-xs text-slate-400 font-medium mb-4">Geriausiems rezultatams rekomenduojame naudoti šį tekstą tiesiogiai į pasirinktą generatorių.</p>
                  <button 
                    onClick={copyToClipboard}
                    className="w-full bg-white text-slate-900 font-black py-5 rounded-2xl flex items-center justify-center gap-3 hover:bg-indigo-50 transition-all active:scale-95 shadow-xl"
                  >
                    KOPY & PARUOŠTI <ArrowRight className="w-5 h-5" />
                  </button>
                </div>
              </div>

              {/* Extra Info */}
              <div className="space-y-4">
                <button 
                  onClick={() => setShowTips(!showTips)}
                  className="w-full bg-white p-5 rounded-2xl border border-slate-200 flex items-center justify-between font-bold text-sm hover:shadow-md transition-all"
                >
                  <span className="flex items-center gap-3 text-indigo-600"><Info className="w-5 h-5" /> Eksperto patarimai</span>
                  {showTips ? <ChevronUp className="w-4 h-4" /> : <ChevronDown className="w-4 h-4" />}
                </button>
                {showTips && (
                  <div className="bg-white p-6 rounded-2xl border border-slate-200 text-sm text-slate-600 space-y-3 shadow-inner">
                    <p>• <strong>Būkite konkretūs:</strong> Vietoj "gražus automobilis" rašykite "matinis juodas visureigis miško kelyje".</p>
                    <p>• <strong>Apšvietimas:</strong> Tai 80% vaizdo nuotaikos. Visada parinkite apšvietimo tipą.</p>
                    <p>• <strong>Teksto erdvė:</strong> Jei planuojate dėti tekstą vėliau, prompt'e nurodykite "clean negative space".</p>
                  </div>
                )}
              </div>

            </div>
          </div>
        </div>

        {/* AI Tools Section */}
        <section className="mt-20">
          <div className="flex flex-col items-center mb-12">
            <h2 className="text-3xl font-black mb-4">Pasirinkite Generatorių</h2>
            <p className="text-slate-500 font-medium text-center max-w-lg">Spustelėkite ant bet kurio įrankio – promptas bus nukopijuotas automatiškai ir atidarytas naujas skirtukas.</p>
          </div>
          <div className="grid grid-cols-2 md:grid-cols-3 lg:grid-cols-6 gap-4">
            {tools.map((tool) => (
              <button
                key={tool.name}
                onClick={() => openTool(tool.url)}
                className="group bg-white p-6 rounded-[2rem] border border-slate-200 hover:border-indigo-500 hover:shadow-xl hover:shadow-indigo-500/10 transition-all text-center flex flex-col items-center"
              >
                <div className="text-3xl mb-3 group-hover:scale-110 transition-transform">{tool.icon}</div>
                <div className="font-black text-sm mb-1">{tool.name}</div>
                <div className="text-[10px] text-slate-400 font-bold uppercase tracking-tighter">{tool.desc}</div>
              </button>
            ))}
          </div>
        </section>
      </main>

      <footer className="max-w-7xl mx-auto px-6 py-12 text-center text-slate-400 text-xs border-t border-slate-200 mt-20">
        <div className="flex items-center justify-center gap-2 mb-4">
           <Zap className="w-4 h-4 text-indigo-500" />
           <span className="font-bold uppercase tracking-widest">DI Generatorius v2.0 Premium</span>
        </div>
        <p>© 2025 Sukurta siekiant aukščiausios kokybės marketingo vizualų.</p>
      </footer>
    </div>
  );
};

export default App;
